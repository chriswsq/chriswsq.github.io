centos使用docker-compose部署zabbix5.2

2021-04-23
version: '3'
services:
 zabbix_mysql:
    image: mysql:5.7
    container_name: zabbix_mysql
    environment:
      - "MYSQL_USER=zabbix"
      - "MYSQL_ROOT_PASSWORD=zabbix_test"
      - "MYSQL_DATABASE=zabbix_test"
    command: --character-set-server=utf8 --collation-server=utf8_bin
    volumes:
      - /etc/localtime:/etc/localtime
      - /bsn/zabbix/mysql:/var/lib/mysql
    networks:
      zabbix:
        ipv4_address: 172.16.0.3
    ports:
      - "3306:3306"
    restart: always

 zabbix-java-gateway:
    image: zabbix/zabbix-java-gateway:centos-5.2-latest
    container_name: zabbix-java-gateway
    networks:
      zabbix:
        ipv4_address: 172.16.0.4
    restart: always

 zabbix-server:
    image: zabbix/zabbix-server-mysql:centos-5.2-latest
    container_name: zabbix_server
    links:
      - zabbix_mysql:mysql
      - zabbix-java-gateway:zabbix-java-gateway
    volumes:
      - /etc/localtime:/etc/localtime
      - /etc/timezone:/etc/timezone
      - /bsn/zabbix/alertscripts:/usr/lib/zabbix/alertscripts
      - /bsn/zabbix/externalscripts:/usr/lib/zabbix/externalscripts
    networks:
      zabbix:
        ipv4_address: 172.16.0.5
    environment:
      - "DB_SERVER_HOST=zabbix_mysql"
      - "MYSQL_DATABASE=zabbix"
      - "MYSQL_USER=zabbix"
      - "MYSQL_PASSWORD=zabbix_test"
      - "MYSQL_ROOT_PASSWORD=zabbix_test"
      - "ZBX_JAVAGATEWAY=zabbix-java-gateway"
    ports:
      - 10051:10051
    restart: always

 zabbix-web:
    container_name: zabbix-web
    image: zabbix/zabbix-web-nginx-mysql:centos-5.2-latest
    networks:
      zabbix:
        ipv4_address: 172.16.0.6
    links:
      - zabbix_mysql:mysql
      - zabbix-java-gateway:zabbix-java-gateway
    environment:
      - "DB_SERVER_HOST=zabbix_mysql"
      - "MYSQL_DATABASE=zabbix"
      - "MYSQL_USER=zabbix"
      - "MYSQL_PASSWORD=zabbix_test"
      - "MYSQL_ROOT_PASSWORD=zabbix_test"
      - "PHP_TZ=Asia/Shanghai"
    ports:
      - 8088:8080
    restart: always
networks:
  zabbix:
    ipam:
      config:
      - subnet: 172.16.0.0/24

#调整zabbix-web乱码

将SIMKAI.ttf 拷贝至 zaabix-web容器 /usr/share/zabbix/assets/fonts/
docker cp ./fonts/SIMKAI.ttf zabbix-web:/usr/share/zabbix/assets/fonts/
docker exec -it zabbix-web bash
vi /usr/share/zabbix/include/defines.inc.php
# 更改前
define('ZBX_GRAPH_FONT_NAME',           'DejaVuSans'); // font file name
# 更改后
define('ZBX_GRAPH_FONT_NAME',           'SIMKAI'); // font file name
